ls -a :: To show all the files and folders including hidden
cd ~ :: takes to root directory 
pwd :: To get the path of current location
cd .. :: to walk back one step in the path
cd ../.. :: Walks back two steps in the path
mkdir <directory name> :: Creates directory
touch <file name> :: Creates the new file of the name specified
open <file name> :: opens the files for mac
code <file name> :: opens the file for windows
open -a /Applications/Visual\ Studio\ Code.app <file name> :: opens the file with the appication mentioned
rm <file name> :: Deletes the file
